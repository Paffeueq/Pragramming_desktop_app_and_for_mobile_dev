\subsection{Narzędzia i wymagania}

\subsubsection{Wymagania systemowe}

\begin{itemize}
    \item \textbf{.NET 8 SDK} lub nowszy
    \item \textbf{Visual Studio 2022} (17.9+) lub \textbf{Visual Studio Code}
    \item \textbf{Docker Desktop} (dla kontenerów)
    \item \textbf{Workload Aspire}:
\end{itemize}

\begin{lstlisting}[language=bash]
dotnet workload install aspire
\end{lstlisting}

\subsubsection{Template'y projektów}

\begin{lstlisting}[language=bash]
# Lista template'ow Aspire
dotnet new list aspire

# Utworzenie nowego projektu
dotnet new aspire-starter -n MyAspireApp
\end{lstlisting}
